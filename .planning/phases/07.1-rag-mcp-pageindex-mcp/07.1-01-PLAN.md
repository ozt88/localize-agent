func extractTermName(english string) string {
    for _, sep := range []string{" - ", " – "} {
        if idx := strings.Index(english, sep); idx > 0 {
            return strings.TrimSpace(english[:idx])
        }
    }
    return strings.TrimSpace(english)
}
