    **Step 3: Word-boundary 매칭 (lore.go 패턴 Python 재구현)**
    ```python
    import re

    def contains_term(text: str, term: str) -> bool:
        if len(term) < 3:
            return False
        suffix = r"(?:'s|s|ed|ing)?"
        pattern = rf"(?i)(?:^|[^A-Za-z0-9]){re.escape(term)}{suffix}(?:[^A-Za-z0-9]|$)"
        return bool(re.search(pattern, text))

    def match_rag_hints(entries: list, batch_text: str, max_hints: int = 3) -> list:
        hits = []
        seen = set()
        for entry in entries:
            terms = [entry["term"]] + entry.get("aliases", [])
            for term in terms:
                term = term.strip()
                if not term or len(term) < 3:
                    continue
                if contains_term(batch_text, term):
                    if entry["term"] not in seen:
                        seen.add(entry["term"])
                        hits.append({
                            "term": entry["term"],
                            "description": entry["description"],
                            "category": entry["category"]
                        })
                    break
        # Sort by term length descending (more specific first)
        hits.sort(key=lambda h: len(h["term"]), reverse=True)
        return hits[:max_hints]
    ```

    **Step 4: 전체 배치 매칭 실행**

    - 결과: `{"batch_id": [{"term": ..., "description": ..., "category": ...}, ...], ...}`
