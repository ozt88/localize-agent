```go
func RunRetranslateSelect(cfg RetranslateSelectConfig) int {
```

```go
func (s *Store) SelectRetranslationBatches(threshold float64, contentType string) ([]contracts.RetranslationCandidate, error) {
    // batch_id에 score_final < threshold인 항목이 1개라도 있는 배치 선택

               MIN(p.score_final) AS min_score,
               AVG(p.score_final) AS avg_score
        FROM pipeline_items_v2 p
        WHERE p.state = 'done'
          AND p.batch_id IN (
              SELECT DISTINCT batch_id FROM pipeline_items_v2
              WHERE state = 'done' AND score_final >= 0 AND score_final < ?
          )
        GROUP BY p.batch_id
        ORDER BY min_score ASC`

}
```

```go
package main

import (
    "flag"
    "fmt"
    "os"

    "localize-agent/workflow/internal/v2pipeline"
    "localize-agent/workflow/pkg/shared"
)

func main() {
    os.Exit(run())
}

func run() int {
    fs := flag.NewFlagSet("go-retranslate-select", flag.ExitOnError)

    var cfg v2pipeline.RetranslateSelectConfig

    fs.StringVar(&cfg.Project, "project", "", "project name (required)")
    fs.Float64Var(&cfg.ScoreThreshold, "score-threshold", 0, "score_final threshold (select items below this)")
    fs.BoolVar(&cfg.DryRun, "dry-run", true, "dry-run mode: show candidates without resetting state")
    fs.BoolVar(&cfg.Histogram, "histogram", false, "show score_final distribution histogram")
    fs.StringVar(&cfg.ContentType, "content-type", "", "filter by content_type (e.g., dialogue)")

    if err := fs.Parse(os.Args[1:]); err != nil {
        fmt.Fprintf(os.Stderr, "parse flags: %v\n", err)
        return 1
    }

    if cfg.Project == "" {
        fmt.Fprintf(os.Stderr, "error: -project is required\n")
        return 1
    }

    // Load project config (same pattern as go-v2-pipeline)
    projCfg, projDir, err := shared.LoadProjectConfig(cfg.Project)
    if err != nil {
        fmt.Fprintf(os.Stderr, "load project config: %v\n", err)
        return 1
    }

    cfg.ProjectDir = projDir
    cfg.CheckpointBackend = projCfg.CheckpointBackend
    cfg.CheckpointDSN = projCfg.CheckpointDSN
    cfg.CheckpointDB = projCfg.CheckpointDB

    // Override from explicit flags
    explicit := make(map[string]bool)
    fs.Visit(func(f *flag.Flag) { explicit[f.Name] = true })

    return v2pipeline.RunRetranslateSelect(cfg)
}
```

주의: `shared.LoadProjectConfig`의 실제 시그니처와 반환값을 read_first에서 확인 후 정확히 맞출 것. 위 코드는 패턴 참조이며, go-v2-pipeline/main.go의 실제 호출 패턴을 그대로 따라야 함.
