1. `export.go`의 `BuildV3Sidecar` 함수에서, entries dedup 로직을 수정한다:
   - 기존: `if !seen[item.SourceRaw]` (first-seen-wins)
   - 변경: source_raw별로 highest RetranslationGen 항목을 추적하는 2-pass 방식
   - Pass 1: `bestGen map[string]int`와 `bestItem map[string]contracts.V2PipelineItem`으로 각 source_raw의 최고 gen 항목을 기록한다. 같은 gen이면 먼저 등장하는 항목(sort_index 순) 우선.
   - Pass 2: 원래 items 순서대로 순회하면서, `bestItem[item.SourceRaw].ID == item.ID`인 경우만 entries에 추가. seen map으로 중복 방지.
   - contextual_entries는 기존과 동일하게 ALL items 순서대로 append.
   - `import "sort"` 불필요 — 2-pass map lookup으로 충분.

구체적인 코드 변경 (export.go BuildV3Sidecar 함수, line 43~104):
```go
func BuildV3Sidecar(items []contracts.V2PipelineItem) V3Sidecar {
    sidecar := V3Sidecar{
        Format:            V3Format,
        Entries:           make([]V3Entry, 0),
        ContextualEntries: make([]V3Entry, 0, len(items)),
    }

    // Pass 1: find highest-gen item for each source_raw (D-02)
    type bestRecord struct {
        gen int
        id  string
    }
    best := make(map[string]bestRecord)
    for _, item := range items {
        if rec, ok := best[item.SourceRaw]; !ok || item.RetranslationGen > rec.gen {
            best[item.SourceRaw] = bestRecord{gen: item.RetranslationGen, id: item.ID}
        }
    }

    // Pass 2: build entries (highest-gen-wins dedup) and contextual_entries (all)
    seen := make(map[string]bool)
    for _, item := range items {
        target := item.KOFormatted
        if target == "" && item.KORaw != "" {
            target = item.KORaw
        }
        target = CleanTarget(target)

        entry := V3Entry{
            ID:          item.ID,
            Source:      item.SourceRaw,
            Target:      target,
            SourceFile:  item.SourceFile,
            TextRole:    item.ContentType,
            SpeakerHint: item.Speaker,
        }
        sidecar.ContextualEntries = append(sidecar.ContextualEntries, entry)

        // Only add to entries if this item is the best (highest-gen) for its source_raw
        if rec := best[item.SourceRaw]; rec.id == item.ID && !seen[item.SourceRaw] {
            seen[item.SourceRaw] = true
            sidecar.Entries = append(sidecar.Entries, entry)
        }
    }

    // Explode multi-line block entries (unchanged)
    // ... existing explode logic ...
}
```

func run() int {
    fs := flag.NewFlagSet("go-v2-reset-all", flag.ExitOnError)
    var (
        project    string
        projectDir string
        backend    string
        dsn        string
        dbPath     string
        dryRun     bool
        cleanup    bool
    )
    fs.StringVar(&project, "project", "", "project name (required)")

    fs.StringVar(&dbPath, "checkpoint-db", "", "SQLite DB path")
    fs.BoolVar(&dryRun, "dry-run", true, "show what would happen without executing")
    fs.BoolVar(&cleanup, "cleanup-stale-claims", true, "reclaim stale working items before reset")

    if err := fs.Parse(os.Args[1:]); err != nil {
        fmt.Fprintf(os.Stderr, "parse flags: %v\n", err)
        return 1
    }
    if project == "" {
        fmt.Fprintln(os.Stderr, "error: -project is required")
        return 1
    }

    // Load project config + explicit flag tracking (go-retranslate-select 패턴)
    projCfg, projDir, err := shared.LoadProjectConfig(project, projectDir)
    // ... explicit flag override logic ...

    // ... error handling ...
    defer store.Close()

    // Pre-reset state
    before, _ := store.CountByState()
    fmt.Printf("Before reset: %v\n", before)

    if dryRun {
        total := before["done"] + before["failed"]
        fmt.Printf("Would reset %d items (done=%d, failed=%d) to pending_translate with retranslation_gen=1\n",
            total, before["done"], before["failed"])
        return 0
    }

    // Cleanup stale claims first

        // store.CleanupStaleClaims() or equivalent
    }

    affected, err := store.ResetAllForRetranslation(1)

    after, _ := store.CountByState()
    fmt.Printf("After reset: %v\n", after)
    fmt.Printf("Reset %d items to pending_translate (retranslation_gen=1)\n", affected)
    return 0
}
